        let assets = [...INITIAL_ASSETS];
        let currentView = 'cards';

        function updateUI() {
            // Update Dashboard Cards
            const grid = document.getElementById('asset-grid');
            grid.innerHTML = assets.map(asset => `
                <div onclick="showDetail(${asset.id})" class="cursor-pointer bg-white p-5 rounded-2xl border border-slate-200 transition-all hover:shadow-xl hover:-translate-y-1">
                    <div class="flex justify-between items-start mb-4">
                        <div class="w-10 h-10 rounded-xl flex items-center justify-center text-white ${asset.color}">${asset.icon}</div>
                        <span class="text-[10px] font-bold text-slate-400 uppercase tracking-widest">${asset.cat}</span>
                    </div>
                    <h3 class="font-bold text-slate-800">${asset.name}</h3>
                    <div class="mt-4 flex items-end gap-1">
                        <span class="text-3xl font-black text-slate-900">${asset.cagr}%</span>
                        <span class="text-slate-400 text-xs mb-1">CAGR</span>
                    </div>
                </div>
            `).join('');

            // Update Table
            const sorted = [...assets].sort((a, b) => b.cagr - a.cagr);
            const tableBody = document.getElementById('ranked-table-body');
            tableBody.innerHTML = sorted.map((asset, index) => `
                <tr class="border-b border-slate-50 hover:bg-slate-50 transition-colors">
                    <td class="px-6 py-4 font-bold text-slate-400">#${index + 1}</td>
                    <td class="px-6 py-4">
                        <div class="flex items-center gap-3">
                            <span class="p-1.5 rounded-lg ${asset.color} text-white text-xs">${asset.icon}</span>
                            <span class="font-bold text-slate-800">${asset.name}</span>
                        </div>
                    </td>
                    <td class="px-6 py-4 text-center text-slate-600">${asset.y1}%</td>
                    <td class="px-6 py-4 text-center text-slate-600">${asset.y2}%</td>
                    <td class="px-6 py-4 text-center text-slate-600">${asset.y3}%</td>
                    <td class="px-6 py-4 text-right font-black text-blue-600">${asset.cagr}%</td>
                </tr>
            `).join('');

            document.getElementById('last-updated').innerText = new Date().toLocaleTimeString();
        }

        function showDetail(id) {
            const asset = assets.find(a => a.id === id);
            const panel = document.getElementById('detail-panel');
            panel.innerHTML = `
                <div class="animate-in fade-in duration-300">
                    <h2 class="text-xl font-bold mb-1">${asset.name}</h2>
                    <span class="text-blue-400 text-xs font-bold uppercase tracking-wider mb-4 block">${asset.cat} Profile</span>
                    
                    <div class="bg-white/10 rounded-2xl p-4 mb-6">
                        <p class="text-slate-300 text-sm leading-relaxed">
                            Projections indicate a 3-year return of <strong>${asset.cagr}%</strong>. 
                            Growth is expected to peak in 2028 at ${asset.y3}% as market conditions stabilize.
                        </p>
                    </div>

                    <div class="space-y-3 text-sm">
                        <div class="flex justify-between border-b border-white/10 pb-2">
                            <span class="text-slate-500">Risk Level</span>
                            <span class="font-semibold text-orange-400">${asset.risk}</span>
                        </div>
                        <div class="flex justify-between border-b border-white/10 pb-2">
                            <span class="text-slate-500">2026 Forecast</span>
                            <span class="font-semibold">${asset.y1}%</span>
                        </div>
                         <div class="flex justify-between">
                            <span class="text-slate-500">2028 Forecast</span>
                            <span class="font-semibold">${asset.y3}%</span>
                        </div>
                    </div>
                </div>
            `;
        }

        function toggleView(view) {
            currentView = view;
            const cardBtn = document.getElementById('view-cards');
            const tableBtn = document.getElementById('view-table');
            const cardCont = document.getElementById('cards-container');
            const tableCont = document.getElementById('table-container');

            if (view === 'cards') {
                cardBtn.className = "px-4 py-2 rounded-lg text-sm font-medium bg-blue-600 text-white";
                tableBtn.className = "px-4 py-2 rounded-lg text-sm font-medium text-slate-600 hover:bg-slate-100";
                cardCont.classList.remove('hidden');
                tableCont.classList.add('hidden');
            } else {
                tableBtn.className = "px-4 py-2 rounded-lg text-sm font-medium bg-blue-600 text-white";
                cardBtn.className = "px-4 py-2 rounded-lg text-sm font-medium text-slate-600 hover:bg-slate-100";
                tableCont.classList.remove('hidden');
                cardCont.classList.add('hidden');
            }
        }

        async function refreshData() {
            const btn = document.getElementById('refresh-btn');
            const icon = document.getElementById('refresh-icon');
            const statusText = document.getElementById('status-text');
            const badge = document.getElementById('status-badge');

            btn.disabled = true;
            icon.classList.add('refresh-spin');
            statusText.innerText = "Syncing Research...";
            badge.classList.add('animate-pulse', 'bg-blue-50', 'text-blue-700');

            // Simulate Network delay with retry pattern
            await new Promise(resolve => setTimeout(resolve, 1500));

            assets = assets.map(a => ({
                ...a,
                cagr: parseFloat((a.cagr + (Math.random() * 0.4 - 0.2)).toFixed(1)),
                y1: parseFloat((a.y1 + (Math.random() * 0.2 - 0.1)).toFixed(1))
            }));

            updateUI();

            btn.disabled = false;
            icon.classList.remove('refresh-spin');
            statusText.innerText = "Research Feed Live";
            badge.classList.remove('animate-pulse', 'bg-blue-50', 'text-blue-700');
        }

        // Initialize
        window.onload = updateUI;
